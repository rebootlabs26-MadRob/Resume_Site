
$lines = @(
    @(@{ Text = '      ______'; Foreground = 'Green' }),
    @(@{ Text = '   .-        -.'; Foreground = 'Blue' }),
    @(@{ Text = ' //            \\'; Foreground = 'Blue' }),
    @(@{ Text = ' |,  .-.  .-.  ,|'; Foreground = 'Blue' }),
    @(
        @{ Text = ' |)('; Foreground = 'Blue' },
        @{ Text = ' _o    o_'; Foreground = 'Red' },
        @{ Text = ' )(|'; Foreground = 'Blue' }
    ),
    @(@{ Text = '  |//   /\   \\|'; Foreground = 'Blue' }),
    @(@{ Text = ' (_     ^^     _)'; Foreground = 'Blue' }),
    @(@{ Text = '  \__|IIIIII|__/'; Foreground = 'White' }),
    @(@{ Text = '   | \IIIIII/ |'; Foreground = 'White' }),
    @(@{ Text = '    \        /'; Foreground = 'Green' }),
    @(@{ Text = '     --------'; Foreground = 'Green' }),
    @(@{ Text = '=== ]:< *Nameless WastedTime* >:[ ==='; Foreground = 'Red'; Background = 'Black' })
)

function Write-ColoredLines {
    param([array]$Lines)

    foreach ($line in $Lines) {
        if ($line -is [System.Array]) {
            foreach ($seg in $line) {
                $text = $seg.Text
                $fg = if ($seg.ContainsKey('Foreground')) { $seg.Foreground } else { $null }
                $bg = if ($seg.ContainsKey('Background')) { $seg.Background } else { $null }
                if ($bg) { Write-Host -NoNewline -ForegroundColor $fg -BackgroundColor $bg $text }
                elseif ($fg) { Write-Host -NoNewline -ForegroundColor $fg $text }
                else { Write-Host -NoNewline $text }
            }
            Write-Host
        }
        elseif ($line -is [System.Collections.Hashtable]) {
            $text = $line.Text
            $fg = if ($line.ContainsKey('Foreground')) { $line.Foreground } else { $null }
            $bg = if ($line.ContainsKey('Background')) { $line.Background } else { $null }
            if ($bg) { Write-Host -ForegroundColor $fg -BackgroundColor $bg $text }
            elseif ($fg) { Write-Host -ForegroundColor $fg $text }
            else { Write-Host $text }
        }
        else {
            Write-Host $line
        }
    }
}

# Display
Write-ColoredLines $lines

# Neon green default text
$Host.UI.RawUI.ForegroundColor = "Green"

function prompt {
    Write-Host "=== " -NoNewline -ForegroundColor Blue
    Write-Host "]:< " -NoNewline -ForegroundColor Red
    Write-Host "*Nameless*" -NoNewline -ForegroundColor DarkYellow
    Write-Host " >:[" -NoNewline -ForegroundColor Red
    Write-Host " ===" -NoNewline -ForegroundColor Blue
    Write-Host ""
    Write-Host "PS $($pwd.Path)> " -NoNewline -ForegroundColor Cyan
    return " "
}
